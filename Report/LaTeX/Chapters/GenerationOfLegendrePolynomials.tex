        double L_plus = 1.0 , L = 0.0, L_minus;
        for (int j = 0; j < n; j++)
        {
        L_minus = L;
        L = L_plus;
        L_plus = (2.0*j +1)*root*L - j*L_minus ;
        L_plus /= j+1;

		//derivative of the Legendre polynomial (L_plus)
        dev_L = (-n*root*L_plus + n*L)/(1-root*root);   
        // Newton's method
